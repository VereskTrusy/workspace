function submitForm() {
    // 폼데이터 담기

    for (var i = 0; i < filesArr.length; i++) {
        // 삭제되지 않은 파일만 폼데이터에 담기
        if (!filesArr[i].is_delete) {
            formData.append("attach_file", filesArr[i]);
        }
    }
    console.log('formData =', formData);
 
    $.ajax({
        method: 'POST',
        url: '/fileppro/Register.do',
        dataType: 'json',
        data: formData,
        enctype : "multipart/form-data", 
        async: true,
        cache: false,
        contentType : false, 
        processData : false,
        success: function () {
            alert("파일업로드 성공");
        },
        error: function (xhr) {
            alert('에러가 발생 하였습니다.');
         
        }
    })
}
